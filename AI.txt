<script>
    lucide.createIcons();
    
    const chatWindow = document.getElementById('chat-window');
    const userInput = document.getElementById('userInput');
    const historyDiv = document.getElementById('chatHistory');

    // Persistence
    let state = {
        name: "Eren",
        theme: "dark",
        persona: "Standard",
        chats: ["Hex Engine Fix", "Pokemon Z Lore", "UI Design"]
    };

    function renderHistory() {
        historyDiv.innerHTML = '';
        state.chats.forEach((chat, idx) => {
            const item = document.createElement('div');
            item.className = 'chat-item';
            item.innerHTML = `<span><i data-lucide="message-circle" style="width:14px; vertical-align:middle; margin-right:8px;"></i>${chat}</span>`;
            
            // Swipe to delete logic
            let startX;
            item.addEventListener('touchstart', e => startX = e.touches[0].clientX);
            item.addEventListener('touchend', e => {
                let endX = e.changedTouches[0].clientX;
                if (startX - endX > 80) {
                    item.classList.add('swiped-out');
                    setTimeout(() => {
                        state.chats.splice(idx, 1);
                        renderHistory();
                    }, 300);
                }
            });
            historyDiv.appendChild(item);
        });
        lucide.createIcons();
    }

    function showSection(section) {
        document.getElementById('home-section').style.display = section === 'home' ? 'block' : 'none';
        document.getElementById('chat-interface').style.display = section === 'chat' ? 'flex' : 'none';
        
        document.querySelectorAll('.nav-item').forEach(el => el.classList.remove('active'));
        event.currentTarget.classList.add('active');
    }

    async function processLithium() {
        const query = userInput.value.trim();
        if (!query) return;

        const mode = document.getElementById('modeSelector').value;
        appendMsg(query, 'user-msg');
        userInput.value = '';

        // Personality Logic
        let response = "";
        let prefix = "<b>Lithium:</b> ";

        if (mode === 'image') {
            appendMsg(prefix + "Synthesizing pixels. Watch this...", 'lithium-msg');
            const imgUrl = `https://pollinations.ai/p/${encodeURIComponent(query)}?width=1080&height=1080&nologo=true`;
            setTimeout(() => {
                appendMsg(`${prefix}Generated Ion for: <i>"${query}"</i><br><img src="${imgUrl}" class="generated-ion">`, 'lithium-msg');
            }, 1000);
            return;
        }

        // Mock AI Responses with Personality
        setTimeout(() => {
            if (mode === 'voltage') {
                response = "Code's clean. I've optimized the logic—run it in the terminal and it should go live instantly. Don't worry about the overhead; I've handled the memory leaks.";
            } else if (mode === 'research') {
                response = `Scouring the web nodes for "${query}"... Synchronizing current data. Here is what's happening right now in 2026.`;
            } else if (mode === 'tutor') {
                response = "Let's break this down. Think of it like a Pokemon battle engine: if the logic doesn't flow, the move fails. Here's the core concept...";
            } else if (mode === 'thought') {
                response = "Deep analysis engaged. Processing multiple outcome vectors... Layer 1 complete. Here is the most logical path.";
            } else {
                response = "I'm on it. Standard processing complete. Anything else for Project Lithium?";
            }
            appendMsg(prefix + response, 'lithium-msg');
        }, 600);
    }

    function appendMsg(text, type) {
        const div = document.createElement('div');
        div.className = `message ${type}`;
        div.innerHTML = text;
        chatWindow.appendChild(div);
        chatWindow.scrollTop = chatWindow.scrollHeight;
    }

    function toggleSettings() {
        const overlay = document.getElementById('settings-overlay');
        const modal = document.getElementById('settings-modal');
        const isVisible = overlay.style.display === 'block';
        overlay.style.display = isVisible ? 'none' : 'block';
        modal.style.display = isVisible ? 'none' : 'block';
    }

    function saveSettings() {
        state.name = document.getElementById('nameInput').value || state.name;
        state.theme = document.getElementById('themeSelect').value;
        state.persona = document.getElementById('personaSelect').value;

        document.getElementById('userNameDisplay').innerText = state.name;
        document.body.setAttribute('data-theme', state.theme);
        document.getElementById('personaTag').innerText = `PERSONA: ${state.persona.toUpperCase()}`;
        
        toggleSettings();
    }

    // Init
    renderHistory();
    userInput.addEventListener('keypress', e => { if(e.key === 'Enter') processLithium(); });
</script>
